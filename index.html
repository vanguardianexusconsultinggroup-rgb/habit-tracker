  <script>

    const tbody = document.querySelector("#habit-table tbody");


    const initialHabits = [

      "Beber 2L de agua",

      "Leer 20 minutos",

      "Hacer ejercicio",

    ];


    function createRow(name) {

      const tr = document.createElement("tr");


      const nameTd = document.createElement("td");

      nameTd.textContent = name;

      tr.appendChild(nameTd);


      for (let i = 0; i < 7; i++) {

        const td = document.createElement("td");

        const checkbox = document.createElement("input");

        checkbox.type = "checkbox";

        checkbox.addEventListener("change", updateProgress);

        td.appendChild(checkbox);

        tr.appendChild(td);

      }


      const progressTd = document.createElement("td");

      progressTd.className = "progress-cell";

      progressTd.textContent = "0%";

      tr.appendChild(progressTd);


      const deleteTd = document.createElement("td");

      const deleteBtn = document.createElement("button");

      deleteBtn.textContent = "✕";

      deleteBtn.className = "delete-btn";

      deleteBtn.onclick = () => tr.remove();

      deleteTd.appendChild(deleteBtn);

      tr.appendChild(deleteTd);


      return tr;

    }


    function updateProgress() {

      const row = this.closest("tr");

      const checkboxes = row.querySelectorAll('input[type="checkbox"]');

      const checked = Array.from(checkboxes).filter(c => c.checked).length;

      const percent = Math.round((checked / 7) * 100);

      const progressCell = row.querySelector(".progress-cell");

      progressCell.textContent = percent + "%";

    }


    function addHabit() {

      const input = document.getElementById("habit-name");

      const name = input.value.trim();

      if (!name) return;

      tbody.appendChild(createRow(name));

      input.value = "";

    }


    // Inicial

    initialHabits.forEach(h => tbody.appendChild(createRow(h)));
